import io
import os
import base64
from textwrap import dedent

import pandas as pd
import matplotlib.pyplot as plt

csv_path = "employees.csv"

sample_csv = (
    "employee_id,department,region,performance_score,years_experience,satisfaction_rating\n"
    "EMP001,R&D,Middle East,91.67,2,3.7\n"
    "EMP002,Operations,Africa,84.83,3,3\n"
    "EMP003,IT,Europe,92.06,14,4.9\n"
    "EMP004,R&D,North America,92.67,5,4.1\n"
    "EMP005,HR,Latin America,86.43,6,3.5\n"
    "EMP006,Sales,Europe,88.50,4,4.0\n"
    "EMP007,Sales,Asia,79.20,7,3.2\n"
    "EMP008,Sales,North America,91.00,5,4.4\n"
    "EMP009,Finance,Asia,85.10,6,3.6\n"
    "EMP010,Marketing,Europe,82.30,3,3.8\n"
)

if os.path.exists(csv_path):
    df = pd.read_csv(csv_path)
else:
    df = pd.read_csv(io.StringIO(sample_csv))

sales_count = (df["department"].astype(str).str.strip() == "Sales").sum()
print(f'Frequency count for "Sales" department: {sales_count}')

dept_counts = df["department"].astype(str).str.strip().value_counts().sort_index()

plt.figure(figsize=(8, 5))
plt.bar(dept_counts.index, dept_counts.values)
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Count")
plt.xticks(rotation=45, ha="right")
plt.tight_layout()
plt.show()
